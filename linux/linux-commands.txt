***LINUX - COMANDOS***

RPM 
	DIRECTORY - /var/lib/rpm

	INSTALL
		- rpm -Uvh <URL>
	UNINSTALL 
		- rpm -e <PACKAGE-NAME>

	FIND QUERY A PACKAGE 
		- rpm -q <PACKAGE-NAME>

ARQUIVO TEXTO

echo <texto> >> <nome-arquivo> (adiona linha no final do arquivo)

echo <texto> > <nome-arquivo> (subscreve tudo que existe pelo texto)
